def list_stats(nums):
    return max(nums), min(nums), sum(nums), sum(nums) / len(nums)

if __name__ == "__main__":
    n = int(input("Enter number of elements: "))
    nums = [float(input()) for _ in range(n)]
    mx, mn, s, avg = list_stats(nums)
    print(f"Max: {mx}, Min: {mn}, Sum: {s}, Avg: {avg}")
